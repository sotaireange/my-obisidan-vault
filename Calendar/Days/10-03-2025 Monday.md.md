const tasks = dv.pages("#tasks")

    .filter(t => !t.completed)
    .map(t => {
        const link = t.file.link;
        const created = t.date;
        const tags = t.tags;
        const priority = t.priority;

        const reviewValue = `x+${points}`;

label: "Finish"
